\hypertarget{projects}{%
\section{Projects}\label{projects}}

\begin{cventries}
    \cventry{Visual Cognition Lab, Department of Psychology, University of Cambridge}{Improving Human Performance in X-ray Baggage Screening}{}{2018 --> 2021}{\begin{cvitems}
\item This project focussed on developing novel training to improve human performance while using the next generation of computed tomography baggage screening technology in airports
\item Collected behavioural data from approximately 400 human research participants with applied visual search tasks programmed in Python and online with the Gorilla platform; conducted statistical analysis and modelling of decision-making, null hypothesis significance testing, Bayesian model comparisons and linear mixed-effects models
\item Won funding from the Defence and Security Accelerator (worth £266,359), worked closely with government and industry stakeholders (e.g., Dstl, QinetiQ), outputs included three published papers and a library of three-dimensional computerised tomography images for training and security research
\end{cvitems}}
    \cventry{Centre for Vision and Cognition, School of Psychology, University of Southampton}{Individual Differences in Dynamic Visual Search}{}{2013 --> 2017}{\begin{cvitems}
\item This project examined the individual cognitive, affective and personality factors that can impact human attention and the relevance of this to complex real-world visual tasks, such as air traffic control
\item I developed experimental tasks to collect behavioural responses, eye movements and physiological data from human participants, modelled human performance using Signal Detection Theory and conducted statistical analysis including linear mixed-effects models
\item Outputs from this project included presentations at national and international conferences, two published papers and project pages available on Github (github.com/alexmuhl-r/Dynamic-Visual-Search) and the Open Science Framework (osf.io/ahufd)
\end{cvitems}}
\end{cventries}
